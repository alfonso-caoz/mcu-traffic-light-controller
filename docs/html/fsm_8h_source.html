66typedef enum {
67 IDLE_VEHICLE_GREEN_PEDESTRIAN_RED, ///< State 0: Idle State waiting for an Interruption. Vehicle has a green light, and Pedestrian a red light. Commute: Button B1/B2 pressed & Time >= 10 s.
68 VEHICLE_FLASHING_GREEN_PEDESTRIAN_RED, ///< State 1: Vehicle has a flashing green light, and Pedestrian a red light. Commute: Time = 3 s. Flashing every 0.5 s.
69 VEHICLE_RED_PEDESTRIAN_RED, ///< State 2: Vehicle and Pedestrian have red lights. Commute: Time = 1 s.
70 VEHICLE_RED_PEDESTRIAN_GREEN, ///< State 3: Vehicle has a red light, and Pedestrian a green light. Commute: Time = 10 s.
71 VEHICLE_RED_PEDESTRIAN_FLASHING_GREEN ///< State 4: Vehicle has a red light, and Pedestrian a flashing green light. Commute: Time = 3 s. Flashing every 0.5 s.
72} TrafficLightState;
